  data: {
    defaultSize: 'default',
    primarySize: 'default',

    disabled: false,
    plain: false,
    loading: false
  },
  setDisabled: function(e) {
    this.setData({
      disabled: !this.data.disabled
    })
  },
  setPlain: function(e) {
    this.setData({
      plain: !this.data.plain
    })
  },
  setLoading: function(e) {
    this.setData({
      loading: !this.data.loading
    })
  }
}

    pageObject[type] = function(e) {
      var key = type + 'Size'
      var changedData = {}
      changedData[key] =

    }
